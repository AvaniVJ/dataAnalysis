# Create a function for a donut chart
def plot_donut_chart(data, title, colors):
    plt.figure(figsize=(8, 8))
    wedges, texts, autotexts = plt.pie(
        data, labels=data.index, autopct='%1.1f%%', colors=colors,
        wedgeprops=dict(width=0.3)  # Creates the donut effect
    )
    plt.setp(autotexts, size=10, weight="bold")
    plt.title(title)
    plt.show()

# Aggregate sales by pizza category and pizza size
category_sales = df.groupby('pizza_category')['total_price'].sum()
size_sales = df.groupby('pizza_size')['total_price'].sum()

# Define color palettes
category_colors = sns.color_palette("pastel", len(category_sales))
size_colors = sns.color_palette("pastel", len(size_sales))

# Plot donut charts
plot_donut_chart(category_sales, "Percentage of Sales by Pizza Category", category_colors)
plot_donut_chart(size_sales, "Percentage of Sales by Pizza Size", size_colors)
